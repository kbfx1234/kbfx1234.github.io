class Solution {

    bool canPartition(vector<int>& nums) {
        int k = nums.size();
        int sum = 0;
        for (auto s : nums) {
            sum += s;
        }
        if (sum % 2 == 1) return false;
        int target = sum / 2;
        vector<vector<bool> > dp(k, vector<bool> (target + 1, false));
        if (nums[0] < target) {
            dp[0][nums[0]] = true;
        }
        for (int i = 1; i < k; i++) {
            for (int j = 0; j <= target; j++) {
                dp[i][j] = dp[i - 1][j];

                if (nums[i] == j) {
                    dp[i][j] = true;
                    continue;
                }

                if (nums[i] < j) {
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - nums[i]];
                }
            }
        }
        return dp[k - 1][target];
              
    }
};

class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int len = people.size();
        sort(people.rbegin(), people.rend());
        int ans = 0;
        int rt = len - 1;
        int lf = 0;
        while(lf <= rt) {
            if ((people[lf] + people[rt]) <= limit) {
                ans += 1;
                lf += 1;
                rt -= 1;
            } 
            else {
                ans += 1;
                lf += 1;
            }  
        }
        return ans;
    }
};
