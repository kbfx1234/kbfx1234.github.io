# 416. 分割等和子集

// cpp 7-01 0-1 背包问题
// 二维dp
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int k = nums.size();
        int sum = 0;
        for (auto s : nums) {
            sum += s;
        }
        if (sum % 2 == 1) return false;
        int target = sum / 2;
        vector<vector<bool> > dp(k, vector<bool> (target + 1, false));
        if (nums[0] < target) {
            dp[0][nums[0]] = true;
        }
        for (int i = 1; i < k; i++) {
            for (int j = 0; j <= target; j++) {
                dp[i][j] = dp[i - 1][j];

                if (nums[i] == j) {
                    dp[i][j] = true;
                    continue;
                }

                if (nums[i] < j) {
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - nums[i]];
                }
            }
        }
        return dp[k - 1][target];
              
    }
};

class Solution {

    bool canPartition(vector<int>& nums) {
        int k = nums.size();
        int sum = 0;
        for (int i : nums) {
            sum += i;
        }

        if (sum % 2 == 1) return false;
        int tar = sum / 2;
        vector<bool> dp(tar + 1, false);
        dp[0] = true;

        if (nums[0] <= tar) {
            dp[nums[0]] = true;
        }

        for (int i = 1; i < k; i++) {
            for (int j = tar; j >= nums[i]; j--) { //mark
                dp[j] = dp[j] || dp[j - nums[i]];
            }
        }
        return dp[tar];
    }
};

# 494. 目标和

// cpp dfs暴力搜索
class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        int len = nums.size();
        return dfs(nums, target, 0, 0);
    }

    int dfs(vector<int> & nums, int target, int k, int cur) {
        if (k == nums.size()) {
            return cur == target ? 1 : 0;
        }
        int ad = dfs(nums,target, k + 1, cur + nums[k]);
        int mi = dfs(nums,target, k + 1, cur - nums[k]);
        return ad + mi;
    }
};

}

# 322. 零钱兑换

// cpp 7-3 背包问题 dp
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        if (amount == 0) return 0;
        int len = coins.size();
        // dp[i] 表示金额为i需要最少的硬币数量
        vector<int> dp(amount + 1);
        // base case
        dp[0] = 0;
        for(int i = 1; i <= amount; i++) {
            dp[i] = 1E5;
            for (int j = 0; j < len; j++) {
                // 遍历硬币面额
                if (coins[j] == i) dp[i] = 1; // if 金额 = 面额 只需1个硬币
                else if (i > coins[j]) {
                    dp[i] = min(dp[i], dp[i - coins[j]] + 1);
                }
            }
        }
        if (dp[amount] == 1E5) return -1;
        return dp[amount];
    }
};

# 518. 零钱兑换 II

// 7-4 cpp dp背包问题区别前一题
class Solution {
public:
    int change(int amount, vector<int>& coins) {
        int len = coins.size();
        // dp[i] 表示 凑成总金额i的所有硬币组合数
        vector<int> dp(amount + 1, 0);
        dp[0] = 1;
      	// 先循环哪个决定了排列数和组合数的问题
        for (int coin : coins) {
            for (int j = coin; j <= amount; j++) {
                dp[j] = dp[j] + dp[j - coin];
            }
        }
        return dp[amount];
    }
};

# 455. 分发饼干

// cpp 7-5 贪心
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int index = s.size() - 1;
        int ans = 0;
        for (int i = g.size() - 1; i >= 0; i--) {
            if (index >= 0 && s[index] >= g[i]) {
                ans++;
                index--;
            }
        }
        return ans;
    }
};

# 435. 无重叠区间

// cpp 7-6 贪心（需要按照结尾的区间排序）
class Solution
{
public:
    int eraseOverlapIntervals(vector<vector<int>> &intervals)
    {
        sort(intervals.begin(), intervals.end(), [](const vector<int> &u, const vector<int> &v) { 
            return u[1] < v[1]; 
        });
        int len = intervals.size();
        int tempr = intervals[0][1];
        int ans = 0;
        for (int i = 1; i < len; i++)
        {
            if (tempr > intervals[i][0])
            {
                ans++;
            }
            else
            {
                tempr = intervals[i][1];
            }
        }
        return ans;
    }
};

# 881. 救生艇

// 7-7 cpp 贪心（最重的和最轻的一起考虑）
class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        int len = people.size();
        sort(people.rbegin(), people.rend());
        int ans = 0;
        int rt = len - 1;
        int lf = 0;
        while(lf <= rt) {
            if ((people[lf] + people[rt]) <= limit) {
                ans += 1;
                lf += 1;
                rt -= 1;
            } 
            else {
                ans += 1;
                lf += 1;
            }  
        }
        return ans;
    }
};

# 932. 漂亮数组

// cpp 7-10 分治
class Solution
{
public:
    unordered_map<int, vector<int>> mp;
    vector<int> beautifulArray(int N)
    {
        return f(N);
    }
    vector<int> f(int N)
    {
        vector<int> ans(N, 0);
        int t = 0;
        if (mp.find(N) != mp.end())
        {
            return mp[N];
        }
        if (N != 1)
        {
            for (auto x : f((N + 1) / 2))
            {
                ans[t++] = 2 * x - 1;
            }
            for (auto x : f(N / 2))
            {
                ans[t++] = 2 * x;
            }
        }
        else
        {
            ans[0] = 1;
        }
        mp[N] = ans;
        return ans;
    }
};

        vector<int >ans;
        unordered_map<int ,int>mp;
        for(int i=0;i<nums.size();i++){
            mp[nums[i]]++;
        }
        for(auto m:mp){
            if(m.second==1)
            ans.push_back(m.first);
        }
        return ans;
    }

// 方法二 位运算
class Solution {
public:
    vector<int> singleNumber(vector<int>& nums) {
        long div = 0;
        int a = 0, b = 0;
        // 最终剩下两个出现一次的数的异或值
        for (int v : nums) {
            div ^= v;
        }
        // 关键一步，通过找到最后一位为1的位，讲这两个数可以分在两组里
        div = div & (-div);
        for (int v : nums) {
            if (v&div) {
                a ^= v;
            }
            else {
                b ^= v;
            }
        }
        return {a, b};
    }
};

# 78. 子集

// 7-12 cpp 
// 方法一：位运算
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        int len = nums.size();
        vector<vector<int> > ans = {};
        for (int i = 0; i < (1 << len); i++) {
            vector<int> temp;
            for (int j = 0; j < len; j++) {
                if (i & (1 << j)) {
                    temp.push_back(nums[j]);
                }
            }
        ans.push_back(temp);
        }
    return ans;
    }
};
// 方法二：回溯
class Solution {
public:
    vector<vector<int> > ans;
    vector<int> path;
    void backtrace (vector<int> & nums, int idx) {
        ans.push_back(path);
        if (idx >= nums.size()) {
            return;
        }
        for (int i = idx; i < nums.size(); i++) {
            path.push_back(nums[i]);
            backtrace(nums, i+1);
            path.pop_back();
        }
    }
    vector<vector<int>> subsets(vector<int>& nums) {
        backtrace(nums, 0);
        return ans;
    }
};

# 547. 省份数量

// 并查集
class Solution {
public:
    const static int n = 256; // 并查集边的数量
    int fa[n];

    // 初始化并查集
    void init(int n) {
        for (int i = 0; i < n; i++) {
            fa[i] = i;
        }
    }
    // find 状态压缩
    int find(int x) {
        // 寻找x的祖先
        if (fa[x] == x)  // 如果x是祖先则返回
            return x;
        else
            return find(fa[x]);  // 如果不是则x的爸爸问x的爷爷
    }
    //将 v->u 这条边加入并查集
    void join(int u, int v) {
        u = find(u);
        v = find(v);
        if (u == v) return;
        fa[v] = u;
    }

    int findCircleNum(vector<vector<int>>& isConnected) {
        int n = isConnected.size();
        init(n);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (isConnected[i][j]) 
                { 
                    join(i,j);
                }
            }
        }
        // 统计
        unordered_set<int> uset;
        for (int i = 0; i < n; i++) {
            uset.insert(find(i));
        }
        return uset.size();
    }
};

# 1319. 连通网络的操作次数

// 并查集 要记忆以下，思路不难
class Solution {
public:
    const static int n = 200000;
    int fa[n];
    // 初始化并查集
    void init(int n) {
        for (int i = 0; i < n; i++) {
            fa[i] = i;
        }
    }
    // find 寻根
    int find(int x) {
        if (x != fa[x]) {
            fa[x] = find(fa[x]);
        }
        return fa[x];
    }
    // 将v->u 这条边加入并查集
    void join(int u, int v) {
        u = find(u);
        v = find(v);
        if (u == v) return ;
        fa[v] = u;
    }

    int makeConnected(int n, vector<vector<int>>& connections) {
        init(n);
        int len = connections.size();
        if (len < n - 1) return -1;
        for (int i = 0; i < len; i++) {
            int a = connections[i][0];
            int b = connections[i][1];
            join(a,b);
        }

        int ans = 0;
        // 判断还剩几个连通域
        for (int i = 0; i < n; i++) {
            if (fa[i] == i) ans++;
        }
        return ans - 1;
    }
};

# 39. 组合总和

// 7-21 cpp 回溯+剪枝
class Solution {
public:
    vector<vector<int>> ans;
    vector<int> temp;
    void backtrack(vector<int> &candidates, int target, int sum, int startidx) {
        // 终止条件
        if (sum == target) {
            ans.push_back(temp);
            return;
        }
        // 剪枝
        for (int i = startidx; i < candidates.size() && sum + candidates[i] <= target; i++) {
            sum += candidates[i];
            temp.push_back(candidates[i]);
          //由于每一个元素可以重复使用，下一轮搜索的起点依然是 i
            backtrack(candidates, target, sum, i);
            // 回溯
            sum -= candidates[i];
            temp.pop_back();
        }
    }
    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        int len = candidates.size();        
        sort(candidates.begin(),candidates.end()); //排序
        backtrack(candidates, target, 0, 0);
        return ans;
    }
};

# 40. 组合总和 II

// cpp 7-21 回溯+剪枝 
class Solution {
public:
    vector<vector<int>> ans;
    vector<int> temp;
    void backtrack(vector<int> &candidates, int target, int sum, int startidx) {
        // 终止条件
        if (sum == target) {
            ans.push_back(temp);
            return;
        }
        // 剪枝
        for (int i = startidx; i < candidates.size() && sum + candidates[i] <= target; i++) {
          // 关键一步剪枝去重复
            if (i > startidx && candidates[i] == candidates[i-1]) continue;
            sum += candidates[i];
            temp.push_back(candidates[i]);
            //由于每一个元素可以重复使用，下一轮搜索的起点依然是 i
            backtrack(candidates, target, sum, i + 1);
            // 回溯
            sum -= candidates[i];
            temp.pop_back();
        }
    }
    vector<vector<int>> combinationSum2(vector<int>& candidates, int target) {
        int len = candidates.size();        
        sort(candidates.begin(),candidates.end()); //排序
        backtrack(candidates, target, 0, 0);
        return ans;
    }
};
