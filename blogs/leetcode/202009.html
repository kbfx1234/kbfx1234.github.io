1. DFS:

#include <iostream>
#include <vector>

class Solution {
public:
    int movingCount(int m, int n, int k) {
		vector<vector<bool>> visited(m, vector<bool>(n, false));//记录该位置是否被访问过
		return dfs(0,0,m,n,k,visited);

	int sumD(int x,int y)
		{// 计算数位和
			int sum = 0;
			while(x > 0){
				sum += x%10;
				x = x/10;
			}
			while(y > 0){
				sum +=y%10;
				y = y/10;

	}
};

  • 代码中dfs函数中的二维数组visited，一定要用传引用参数！！！

  • 因为visited是一个普通二维数组（不是动态数组），所以如果不加'&'，就会是以pass by value的形式传进来，那么不管dfs函数体里面怎么操作visited数组，它的值都不会被真正改变到。

  • 所以必须把visited数组的内存地址传进来，这样每次访问完一个元素，把visited[i][j]的值设为true，才能真正改变visited[i][j]的值。

复杂度分析：

M,N 分别代表矩阵行列大小

  • 时间复杂度O(MN)O(MN)O(MN): 最差情况下，机器人遍历矩阵所有单元格，此时时间复杂度为O(MN)O(MN)O(MN)
  • **空间复杂度O(MN)O(MN)O(MN): ** 最差情况下，Set visited 内存储矩阵所有单元格的索引，使用O(MN)O(MN)O(MN)的额外空间。

2.BFS:

DFS是沿着一个方向一直往下走，直到不满足条件才会回头。而BFS就显得有点博爱了，他不是一条道走下去，他会把离他最近的都访问一遍，访问完之后才开始访问第二近的……，一直这样下去，所以最好的一种数据结构就是使用 队列，因为队列是先进先出，离他最近的访问完之后加入到队列中，最先入队的也是最先出队的。

#include <iostream>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
using namespace std;
 
class Solution {
public:
    int movingCount(int m, int n, int k) {
		vector<vector<bool>> visited(m, vector<bool>(n, false));//记录该位置是否被访问
		queue<pair<int, int>> q; //创建一个队列，保存的是访问到的格子坐标，是个二维数组
		int res = 0;
		q.push({0,0});// 第一个坐标先加入队列
		while(q.size() > 0)
		{
			pair<int,int> x = q.front(); // 表示的是提取队列头部元素，因为队列是先进先出，从尾部添加，从头部移除，pop用来移除
			int i = x.first, j = x.second;// pair.fist/second 用来访问pair中的第一第二个元素
			q.pop();
			if (i >= m || j >= n || k < sumD(i, j) || visited[i][j]) continue;
      // 如果越界，或者和大于k，或者已被访问过了，返回0
			visited[i][j] = true;//标记已访问过了
			res++;// 统计满足的坐标
      //把当前格子下边格子的坐标加入到队列中
			q.push({i+1,j});
      //把当前格子右边格子的坐标加入到队列中
			q.push({i,j+1});

		}
		cout<<res<<endl;
		return res;
    }
  // 计算数位和
	int sumD(int x,int y)
		{
			int sum = 0;
			while(x > 0){
				sum += x%10;
				x = x/10;
			}
			while(y > 0){
				sum +=y%10;
				y = y/10;
			}
		return sum;
		}

};

	int m = 24, n = 37, k = 12;

# 剑指 Offer 14- I. 剪绳子

  1. 数学方法：

    # 切分规则：
    1. 最优： 3 。把绳子尽可能切为多个长度为 3 的片段，留下的最后一段绳子的长度可能为 0,1,2 三种情况。
    2. 次优： 2。若最后一段绳子长度为 2 ；则保留，不再拆为1+1 。
    3. 最差： 1 。若最后一段绳子长度为 1 ；则应把一份 3 + 1替换为 2 + 2，因为 2×2>3×1。
class Solution {
public:
    int cuttingRope(int n) {
        if(n <= 3) return n-1;
        int m = n/3, k = n%3;
        if(k == 0) return pow(3,m);
        else if(k == 1) return pow(3,m-1)*4;
        else return pow(3,m)*2;
    }
};

class Solution {
public:
    int cutRope(int number) {    
        vector<int> dp(number+1, -1);
			  dp[0] = 1;
        for(int i=1; i <= number; ++i){
            for(int j=1; j <= i; ++j){
                dp[i] = max(dp[i],j*dp[i-j]);
            }
        }
        return dp[number];
    }
};

class Solution {
private:
        static constexpr int mod = 1000000007;
public:
    int cuttingRope(int n) {
        if(n <= 3) return n-1;
        if(n == 4)  return n;
    
        long ans = 1;
        while(n > 4)
        {
            ans *= 3;
            ans %= mod;
            n -= 3;
        }
		     // 最后n的值只有可能是：2、3、4。而2、3、4能得到的最大乘积恰恰就是自身值
        // 因为2、3不需要再剪了（剪了反而变小）；4剪成2x2是最大的，2x2恰巧等于4

        return ans * n % mod;
    }
};

# 剑指 Offer 15. 二进制中1的个数

9-9

# 方法一：逐位判断

  • 根据 与运算 定义，设二进制数字 nn ，则有：

    • 若 n & 1 = 0 ，则 n二进制 最右一位 为 0 ；
    • 若 n & 1 = 1 ，则 n 二进制 最右一位 为 1 。

  • 根据以上特点，考虑以下 循环判断 ：

  1. 判断 n 最右一位是否为 1 ，根据结果计数。
  2. 将 n 右移一位（本题要求把数字 nn 看作无符号数，因此使用 无符号右移 操作）。
class Solution {
public:
    int hammingWeight(uint32_t n) {
		int count = 0;
		while ((n != 0))
		{
			if(n & 1) count++;
			n >>= 1; //右移一位
		}
		cout<<count<<endl;
		return count;	
    }
};

 
int main()
{
	uint32_t n = 24;
	class Solution Solu;
	Solu.hammingWeight(n);
	return 0;
}

# 方法二：巧用n&(n-1)

  • (n−1) 解析： 二进制数字 n 最右边的 1 变成 0 ，此 1 右边的 0 都变成 1 。
  • n & (n - 1) 解析： 二进制数字 n 最右边的 1 变成 0 ，其余不变。
class Solution {
public:
    int hammingWeight(uint32_t n) {
		int count = 0;
		while ((n != 0))
		{
			count++;
			n &= n - 1;
		}
		cout<<count<<endl;
		return count;	
    }
};

    1. n & 1（与操作）： 判断 n 二进制最右一位是否为 1 ；
    2. n>>1（移位操作）： n 右移一位（可理解为删除最后一位）。
class Solution {
public:
    double myPow(double x, int n) {
		if(x == 0) return 0;
        if(n == 0) return 1;
		long b = n; // 防止爆int
		double ans = 1.0;
		if (b < 0){
			x = 1/x;
			b = -b;
		}
		while(b > 0){
			if((b & 1) == 1) ans *= x;
			x *= x;
			b >>= 1;
		}
		
	// cout<<fixed<<setprecision(5)<<ans<<endl;
    return ans;
	}
};

#include <iostream>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
#include<math.h>
#include<iomanip>
using namespace std;
 

struct ListNode {
      int val;
      ListNode *next;
      ListNode(int x) : val(x), next(NULL) {}
 };
 
class Solution {
public:
    ListNode* deleteNode(ListNode* head, int val) {
		if(head->val == val)	return head->next;
		ListNode *pre = head,   *cur = head->next;
		while(cur != NULL && cur->val != val){
			pre = cur;
			cur = cur->next;
		}
		pre->next = cur->next;
		return head;
    
    }
};
 
int main()

	int n = -3;
	class Solution Solu;
	Solu.myPow(x,n);
	return 0;

首尾双指针

  • 定义头指针 left ，尾指针 right .
  • left一直往右移，直到它指向的值为偶数
  • right一直往左移， 直到它指向的值为奇数
  • 交换 nums[left] 和 nums[right].
  • 重复上述操作，直到 left == right.
class Solution
{
public:
	vector<int> exchange(vector<int> &nums)
	{
		int left = 0, right = nums.size() - 1;
		while (left < right)
		{
			if (nums[left] % 2 != 0)
			{
				left++;
				continue;
			}
			if (nums[right] % 2 != 1)
			{
				right--;
				continue;
			}
			swap(nums[left++], nums[right--]);
		}
		for (auto var : nums)
			cout << var << endl;
		return nums;
	}
};

快慢双指针

  • 定义快慢双指针 fast 和 low ，fast 在前， low在后 .
  • fast 的作用是向前搜索奇数位置，low 的作用是指向下一个奇数应当存放的位置
  • fast向前移动，当它搜索到奇数时，将它和 nums[low] 交换，此时 low 向前移动一个位置
  • 重复上述操作，直到 fast 指向数组末尾 .
class Solution
{
public:
	vector<int> exchange(vector<int> &nums)
	{
		int fast = 0, low = 0;
		while(fast < nums.size())
		{
			if(nums[fast] % 2 == 1)
			{
				swap(nums[low], nums[fast]);
				low++;
			}
			fast++;
		}

		//for (auto var : nums)
			//cout << var << endl;
		return nums;
	}
};

class Solution {

			head = head->next;

    {
        k++;
        L=L->next;

}

快慢指针法。基本思路：

  • 倒数第 k 个节点，其实就是正数第 n - k + 1 个节点。但是我们不知道 n 的值，所以没办法用一个指针来完成任务。
  • 定义一快一慢两个指针，初始都指向 head。然后快指针先从 head 处往前走 k - 1 步，慢指针全程不动。这样快指针走完 k - 1 步之后，它们之间的距离是 k - 1。
  • 然后快慢指针同时开始往前走，当快指针到达链表最后一个元素的时候，二者停下。此时快指针加上之前的一共走了 n 步，而慢指针与快指针之间的距离始终为 k - 1，所以慢指针走了 n - (k - 1)步，即 n - k + 1步，刚好是倒数第 k 个节点的位置。
class Solution {
public:
    ListNode* getKthFromEnd(ListNode* head, int k) {
		ListNode *fast = head, *low = head;
		if (head == NULL) return NULL;
		for (int i = 0; i < k; ++i)
		{
			fast = fast->next;
		}
		while(fast != NULL)
		{
			fast = fast->next;
			low = low->next;
		}
		return low;
	}

};

# 剑指 Offer 25. 合并两个排序的链表

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
		ListNode *ans = new ListNode(0), *cur = ans;
		while(l1 && l2)
		{
			if(l2->val >= l1->val)
			{
				cur->next = l1;
				l1 = l1->next;
			}
			else
			{
				cur->next = l2;
				l2 = l2->next;
			}
			cur = cur->next;
			
		}
		cur->next = l1 !=NULL ? l1 : l2;
		return ans->next;
    }
};

# 剑指 Offer 26. 树的子结构

9-22

struct TreeNode {
      int val;
      TreeNode *left;
      TreeNode *right;
      TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};

class Solution {
public:
    bool isSubStructure(TreeNode* A, TreeNode* B) {
        if(A == NULL || B == NULL) return false;
        // 如果A = B 那就迭代继续比较它们对应的左孩子和右孩子
        // 如果A != B 那就比较A的左孩子和B 或者A 的右孩子和B 
        else return recur(A, B) || isSubStructure(A->left, B) || isSubStructure(A->right, B);
    }
    bool recur(TreeNode* A, TreeNode* B) {
        // 如果 B 已经遍历完了，true
        if (B == NULL) return true;
        // 如果 B 还有，但 A 遍历完了，那 B 剩下的就没法匹配了，false
        if (A == NULL ) return false;
         // 不相等，false
        if (A->val != B->val) return false;
        //当前节点比较完之后还要继续判断左右子节点
        return recur(A->left, B->left) && recur(A->right, B->right);
    }
};

//递归
class Solution {
public:
    TreeNode* mirrorTree(TreeNode* root) {
        if(root = NULL) return NULL;
        TreeNode* tmp = root->left;
        root->left = mirrorTree(root->right);
        root->right = mirrorTree(tmp);
        return root;
    }
};	

// other's solution
class Solution {
public:
    TreeNode* mirrorTree(TreeNode* root) {
        if (root == NULL) {
            return NULL;
        }
        swap(root->left, root->right);
        mirrorTree(root->left);
        mirrorTree(root->right);
        return root;
    }
};
