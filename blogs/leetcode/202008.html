class Solution {
public:
    int minArray(vector<int>& numbers) {
        if(numbers.size()==0) return 0;
        if(numbers.size()==1) return numbers[0];
        int i = 0, j = numbers.size() - 1;
        while(i < j){
            int m = (i + j) / 2;
            if(numbers[m] > numbers[j]) i = m + 1;
            else if(numbers[m] < numbers[j]) j = m;
            else j--;
        }
        return numbers[i];
    }
};
