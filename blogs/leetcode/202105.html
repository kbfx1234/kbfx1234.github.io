// 5-24 BFS
class Solution {
public:
    int sumNumbers(TreeNode* root) {
        // å±‚åºéå†
        int ans = 0;
        queue<TreeNode *> que;
        if (root != NULL) que.push(root);
        while (!que.empty()) {
            TreeNode *Node = que.front();
            que.pop();
            if (Node->left == NULL && Node->right == NULL) {
                ans += Node->val; // å¶ç»“ç‚¹æ±‚å’Œ
            }
            if (Node->left) {
                Node->left->val += Node->val * 10; //æ¯ä¸€å±‚æ±‚å’Œç´¯åŠ 
                que.push(Node->left);
            }
            if (Node->right) {
                Node->right->val += Node->val * 10;
                que.push(Node->right);
            }
            }
            return ans;
        }
};

// DFS
class Solution {
public:
    int sumNumbers(TreeNode* root) {
        return dfs(root, 0);
    }

    int dfs(TreeNode* root, int presum) {
        if (root == NULL) return 0;
        int sum = presum*10 + root->val;
        if (root->left == NULL && root->right == NULL) return sum;
        else return dfs(root->left, sum) + dfs(root->right, sum);
    }
};

class Solution {
public:
    int findBottomLeftValue(TreeNode* root) {
        queue<TreeNode *> que;
        que.push(root);
        int ans = 0;
        while (!que.empty()) {
            int len = que.size();
            for (int i = 0; i < len; i++) {
                TreeNode *Node = que.front();
                if (i == 0) ans = Node->val;
                que.pop();
                if (Node->left) que.push(Node->left);
                if (Node->right) que.push(Node->right);
            }
        }
        return ans;
    }
};

class Codec {
public:
    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        if (root == NULL) return "";
        queue<TreeNode *> que;
        que.push(root);
        string ans;
        while (!que.empty()) {
            int len = que.size();
            for (int i = 0; i < len; i++) {
                TreeNode * Node = que.front();
                que.pop(); 
                if (Node) {
                    ans += to_string(Node->val) + " ";
                    que.push(Node->left);
                    que.push(Node->right);
                }
                else ans += "# ";       
            }
        }
        return ans;   
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        if (data.empty()) return NULL;
        stringstream ss(data);
        string var;
        ss >> var;
        TreeNode * root = new TreeNode(stoi(var));
        queue<TreeNode*> que;
        que.push(root);
        while (!que.empty()) {
            int len = que.size();
            for (int i = 0; i < len; i++) {
                TreeNode * node = que.front();
                que.pop();
                string left, right;
                ss>>left>>right;
                if (left != "#") {
                    TreeNode *left_node = new TreeNode(stoi(left));
                    node->left = left_node;
                    que.push(left_node);
                }
                if (right != "#") {
                    TreeNode *right_node = new TreeNode(stoi(right));
                    node->right = right_node;
                    que.push(right_node);
                }
            }

        }
        return root;
    }
};

        if (root == NULL) return;
      // important
        mp[y].push_back({x, root->val});
        if (root->left) dfs(root->left, x + 1, y - 1);

class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int len = nums.size();
        vector<int> ans(2,-1);
        map<int,int> mp;
        for (int i = 0 ;i < len; i++) {
            if (mp.count(target - nums[i]) > 0) {
                ans[0] = i;
                ans[1] = mp[target - nums[i]];
                break;
            }
            mp[nums[i]] = i;
        }
        return ans;

    }
};

            }
            }

    }
