# 129. 求根节点到叶节点数字之和

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
// 5-24 BFS
class Solution {
public:
    int sumNumbers(TreeNode* root) {
        // 层序遍历
        int ans = 0;
        queue<TreeNode *> que;
        if (root != NULL) que.push(root);
        while (!que.empty()) {
            TreeNode *Node = que.front();
            que.pop();
            if (Node->left == NULL && Node->right == NULL) {
                ans += Node->val; // 叶结点求和
            }
            if (Node->left) {
                Node->left->val += Node->val * 10; //每一层求和累加
                que.push(Node->left);
            }
            if (Node->right) {
                Node->right->val += Node->val * 10;
                que.push(Node->right);
            }
            }
            return ans;
        }
};

// DFS
class Solution {
public:
    int sumNumbers(TreeNode* root) {
        return dfs(root, 0);
    }

    int dfs(TreeNode* root, int presum) {
        if (root == NULL) return 0;
        int sum = presum*10 + root->val;
        if (root->left == NULL && root->right == NULL) return sum;
        else return dfs(root->left, sum) + dfs(root->right, sum);
    }
};

# 513. 找树左下角的值

// 5-25 层次遍历 BFS
class Solution {
public:
    int findBottomLeftValue(TreeNode* root) {
        queue<TreeNode *> que;
        que.push(root);
        int ans = 0;
        while (!que.empty()) {
            int len = que.size();
            for (int i = 0; i < len; i++) {
                TreeNode *Node = que.front();
                if (i == 0) ans = Node->val;
                que.pop();
                if (Node->left) que.push(Node->left);
                if (Node->right) que.push(Node->right);
            }
        }
        return ans;
    }
};

# 297. 二叉树的序列化与反序列化

// 5-26 层序遍历 c++
// stringstream 用的不熟练，要在看看
class Codec {
public:
    // Encodes a tree to a single string.
    string serialize(TreeNode* root) {
        if (root == NULL) return "";
        queue<TreeNode *> que;
        que.push(root);
        string ans;
        while (!que.empty()) {
            int len = que.size();
            for (int i = 0; i < len; i++) {
                TreeNode * Node = que.front();
                que.pop(); 
                if (Node) {
                    ans += to_string(Node->val) + " ";
                    que.push(Node->left);
                    que.push(Node->right);
                }
                else ans += "# ";       
            }
        }
        return ans;   
    }

    // Decodes your encoded data to tree.
    TreeNode* deserialize(string data) {
        if (data.empty()) return NULL;
        stringstream ss(data);
        string var;
        ss >> var;
        TreeNode * root = new TreeNode(stoi(var));
        queue<TreeNode*> que;
        que.push(root);
        while (!que.empty()) {
            int len = que.size();
            for (int i = 0; i < len; i++) {
                TreeNode * node = que.front();
                que.pop();
                string left, right;
                ss>>left>>right;
                if (left != "#") {
                    TreeNode *left_node = new TreeNode(stoi(left));
                    node->left = left_node;
                    que.push(left_node);
                }
                if (right != "#") {
                    TreeNode *right_node = new TreeNode(stoi(right));
                    node->right = right_node;
                    que.push(right_node);
                }
            }

        }
        return root;
    }
};

# 987. 二叉树的垂序遍历

// 5-27 DFS CPP 关键是map 里的顺序将列作为key 
class Solution {
public:
    map<int, vector<pair<int, int> > >mp; // mp(y, x, val)
    void dfs(TreeNode * root, int x, int y) {
        if (root == NULL) return;
      // important
        mp[y].push_back({x, root->val});
        if (root->left) dfs(root->left, x + 1, y - 1);
        if (root->right) dfs(root->right, x + 1, y + 1);
    }
    vector<vector<int>> verticalTraversal(TreeNode* root) {
        dfs(root, 0, 0);
        vector<vector<int>> ans;
        for (auto &p : mp) {
            sort(p.second.begin(), p.second.end()); //先按照行号排序，再按照val排序
            vector<int> v;
            for (auto &q : p.second) {
                v.push_back(q.second);
            }
            ans.push_back(v);
        }
        return ans;
    }
};

# 1. 两数之和

// CPP 5-28 一遍哈希
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int len = nums.size();
        vector<int> ans(2,-1);
        map<int,int> mp;
        for (int i = 0 ;i < len; i++) {
            if (mp.count(target - nums[i]) > 0) {
                ans[0] = i;
                ans[1] = mp[target - nums[i]];
                break;
            }
            mp[nums[i]] = i;
        }
        return ans;

    }
};

# 347. 前 K 个高频元素

// CPP 5-29 哈希 sort 空了要看看堆排序了@@
class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        map<int,int> mp;
        int len = nums.size();
        vector<int> ans(k,0);
        for (int i = 0; i < len; i++) {
            mp[nums[i]]++;
        }
        vector<pair<int, int>> tmp;
        for (auto &p : mp) {
            // 将出现的次数放在第一个值，便于排序
            tmp.push_back({p.second, p.first});
        }
        // sort函数是由多个排序算法封闭的，主要是快排
        sort(tmp.rbegin(),tmp.rend());
        for (int i = 0; i < k; i++) {
            ans[i] = tmp[i].second;
        }
        return ans;
    }
};

# 3. 无重复字符的最长子串

// cpp 5-31 滑动窗口双循环 时间复杂度O(n2) 复杂了。。。
class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {
        int len = s.size();
        int lf = 0, rt = 0;
        string st;
        int ans = 0;
        for (int lf = 0; lf < len; lf++)
        {
            while (rt < len)
            {
                size_t found = st.find(s[rt]);
                if (found != st.npos) {
                    st.erase(0, 1);
                    break;
            } 
            else {
                    st.push_back(s[rt]);
                    rt++;
            }
            }
            ans = max(ans, rt - lf);
        }
        return ans;
    }
};
// 哈希map 时间复杂度O(n)
class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {
        int len = s.size();
        int lf = 0, rt = 0, length = 0;
        unordered_map<char, int> mp;
        int ans = 0;
        while (rt < len) {
            char tmp = s[rt];
            if (mp.find(tmp) != mp.end() && mp[tmp] >= lf) {
                lf = mp[tmp] + 1;
                length = rt - lf;
            }
            mp[tmp] = rt;
            rt++;
            length++;
            ans = max(ans, length);
        }
        return ans;
    }
};
// 简化
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int res = 0;
        int start = 0;
        unordered_map<char, int> map;
        for(int i = 0; i < s.size(); i++) {
            auto c = s[i];
            if(map[c]) {
                start = max(start, map[c]);
            }
            res = max(res, i - start + 1);
            map[c] = i;
        }
        return res;
    }
};
